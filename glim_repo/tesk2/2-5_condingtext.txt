#include <string>
#include <vector>

using namespace std;

int solution(vector<vector<int>> board, int k) {
    int answer = 0;
    
    int n = board.size();
    int m = board[0].size();
    
    for (int j = 0; j < n; j++) 
    {
        for (int i = 0; i < m; i++) 
        {
            if (i + j <= k) 
            {  
                answer += board[j][i];
            }
        }
    }
    
    
    return answer;
}
