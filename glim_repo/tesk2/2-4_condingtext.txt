#include <string>
#include <vector>

using namespace std;

int solution(int a, int b) {
    int answer = 0;
    
    string A = to_string(a);
    string B = to_string(b);
    
    int AB = stoi(A+B);
    int BA = stoi(B+A);
    
    
    answer = max(AB,BA);
    
    return answer;
}
